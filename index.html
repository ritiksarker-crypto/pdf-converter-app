Use these code snippets to convert HTML to PDF programmatically from any application.

JavaScript 
// Client-side HTML to PDF (using html2pdf.js)
// Include: <script src="https://cdnjs.cloudflare.com/ajax/libs/html2pdf.js/0.10.2/html2pdf.bundle.min.js"></script>

const htmlContent = `
  <h1>My Document</h1>
  <p>Generated via API call</p>
`;

// Create a temporary container
const container = document.createElement('div');
container.innerHTML = htmlContent;
document.body.appendChild(container);

// Generate PDF
html2pdf()
  .set({
    margin: 10,
    filename: 'document.pdf',
    image: { type: 'jpeg', quality: 0.98 },
    html2canvas: { scale: 2, useCORS: true },
    jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' }
  })
  .from(container)
  .save()
  .then(() => container.remove());
Python 
# Free: Using weasyprint (pip install weasyprint)
from weasyprint import HTML

html_content = """
<h1>My Document</h1>
<p>Generated with Python</p>
"""

HTML(string=html_content).write_pdf('output.pdf')
print("PDF created successfully!")
